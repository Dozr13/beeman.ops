Pull status from port 4028 for ALL miners (prints one line per miner)

This hits the miners’ API and prints the first ~250 chars of the response (enough to confirm stats/summary output).

python3 - <<'PY'
import socket

ips = [
"192.168.1.101","192.168.1.102","192.168.1.103","192.168.1.104","192.168.1.105","192.168.1.106","192.168.1.107","192.168.1.108","192.168.1.109","192.168.1.110",
"192.168.1.128","192.168.1.133","192.168.1.135","192.168.1.136","192.168.1.150","192.168.1.151","192.168.1.156","192.168.1.161","192.168.1.166","192.168.1.170",
"192.168.1.172","192.168.1.173","192.168.1.176","192.168.1.182","192.168.1.183","192.168.1.197","192.168.1.198","192.168.1.200","192.168.1.203","192.168.1.204",
"192.168.1.207","192.168.1.209","192.168.1.212","192.168.1.217","192.168.1.221","192.168.1.224","192.168.1.225","192.168.1.226","192.168.1.229","192.168.1.232",
"192.168.1.233","192.168.1.237","192.168.1.239","192.168.1.247","192.168.1.248",
]

def query(ip: str, command: str = "summary") -> str:
s = socket.socket()
s.settimeout(2)
s.connect((ip, 4028))
s.send(f'{{"command":"{cmd}"}}'.encode())
data = s.recv(65535)
s.close()
return data.decode(errors="ignore").strip()

for ip in ips:
try:
out = query(ip, "summary")
print(f"{ip}\t{out[:250]}")
except Exception as e:
print(f"{ip}\tERROR\t{e}")
